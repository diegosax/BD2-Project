CREATE TABLE service_doctor
(
  doctor_id integer NOT NULL,
  service_id integer NOT NULL,
  CONSTRAINT service_doctor_pkey PRIMARY KEY (doctor_id, service_id),
  CONSTRAINT fk_doctor FOREIGN KEY (doctor_id)
      REFERENCES "User" (id) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION,
  CONSTRAINT fk_service FOREIGN KEY (service_id)
      REFERENCES service (id) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION
);
