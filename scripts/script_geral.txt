CREATE DATABASE doctor
  WITH OWNER = postgres
       ENCODING = 'UTF8'
       TABLESPACE = pg_default
       LC_COLLATE = 'pt_BR.UTF-8'
       LC_CTYPE = 'pt_BR.UTF-8'
       CONNECTION LIMIT = -1;

CREATE TABLE "User"
(
  id serial NOT NULL,
  "name" text NOT NULL,
  email text NOT NULL,
  "password" text NOT NULL,
  crm text,
  "type" smallint NOT NULL DEFAULT 0,
  CONSTRAINT id PRIMARY KEY (id)
);

CREATE TABLE service
(
  id serial NOT NULL,
  "name" text NOT NULL,
  CONSTRAINT pk_service PRIMARY KEY (id)
);

CREATE TABLE service_doctor
(
  doctor_id integer NOT NULL,
  service_id integer NOT NULL,
  CONSTRAINT service_doctor_pkey PRIMARY KEY (doctor_id, service_id),
  CONSTRAINT fk_doctor FOREIGN KEY (doctor_id)
      REFERENCES "User" (id) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION,
  CONSTRAINT fk_service FOREIGN KEY (service_id)
      REFERENCES service (id) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION
);

CREATE TABLE meeting
(
  id serial NOT NULL,
  notes text,
  "scheduledDate" date NOT NULL,
  "attendedDate" date,
  doctor_id integer NOT NULL,
  client_id integer NOT NULL,
  service_id integer NOT NULL,
  CONSTRAINT pk_meeting PRIMARY KEY (id),
  CONSTRAINT fk_client FOREIGN KEY (client_id)
      REFERENCES "User" (id) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION,
  CONSTRAINT fk_service_doctor FOREIGN KEY (service_id, doctor_id)
      REFERENCES service_doctor (service_id, doctor_id) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION
);
